const express = require('express');
const router = express.Router();
const Setting = require('../models/Setting');
const authenticateToken = require('../middleware/auth');

// GET pincodes
router.get('/', async (req, res) => {
  try {
    let settings = await Setting.findOne();
    if (!settings) {
      settings = await Setting.create({});
    }
    res.json({
      servicePincodes: settings.servicePincodes,
      pincodeCharges: Object.fromEntries(settings.pincodeCharges)
    });
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch pincodes' });
  }
});

// UPDATE pincodes (admin only)
router.put('/', authenticateToken, async (req, res) => {
  try {
    const { servicePincodes, pincodeCharges } = req.body;
    
    let settings = await Setting.findOne();
    if (!settings) {
      settings = new Setting();
    }

    settings.servicePincodes = servicePincodes;
    settings.pincodeCharges = new Map(Object.entries(pincodeCharges));
    
    await settings.save();
    
    res.json({
      servicePincodes: settings.servicePincodes,
      pincodeCharges: Object.fromEntries(settings.pincodeCharges)
    });
  } catch (error) {
    res.status(400).json({ error: 'Failed to update pincodes' });
  }
});

module.exports = router;
